Please modify my existing Solana pump.fun paper trading application by adding a new Token Detail Page.

The goal is to change the user flow: instead of opening a trade modal immediately, clicking a token should now navigate the user to a dedicated page for that token. This new page will display an embedded price chart from gmgn.cc and have "Buy" and "Sell" buttons underneath it, which will then open the trade modal.

// FILE: frontend/src/pages/TokenPage.js

import React, { useState, useEffect } from 'react';
import { useParams, useLocation, Link } from 'react-router-dom';
import { trades } from '../api';
import TradeModal from '../components/TradeModal';

const TokenPage = () => {
  const { tokenAddress } = useParams();
  const location = useLocation();
  
  // The token object is passed via navigation state for efficiency
  const [token, setToken] = useState(location.state?.token || null);
  const [userPosition, setUserPosition] = useState(null);
  const [showModal, setShowModal] = useState(false);
  const [tradeMode, setTradeMode] = useState('buy'); // 'buy' or 'sell'

  useEffect(() => {
    // Fetch user's open positions to check if they own this token
    const fetchPosition = async () => {
      try {
        const res = await trades.getPositions();
        const foundPosition = res.data.positions.find(p => p.token_address === tokenAddress);
        setUserPosition(foundPosition || null);
      } catch (error) {
        console.error("Failed to fetch user positions:", error);
      }
    };

    if (tokenAddress) {
      fetchPosition();
    }
    
    // A fallback could be added here to fetch token data if `location.state` is null
    // For now, we assume navigation provides the token object.

  }, [tokenAddress]);

  const openTradeModal = (mode) => {
    setTradeMode(mode);
    setShowModal(true);
  };

  if (!token) {
    return (
      <div style={styles.container}>
        <h1 style={styles.title}>Loading Token...</h1>
        <p style={styles.subtitle}>If this takes too long, the token might not be available. <Link to="/" style={styles.link}>Go back home.</Link></p>
      </div>
    );
  }

  return (
    <div style={styles.container}>
      <div style={styles.header}>
        <h1 style={styles.title}>{token.name} ({token.symbol})</h1>
        <p style={styles.subtitle}>{token.tokenAddress}</p>
      </div>

      <div style={styles.chartContainer}>
        <iframe
          title={`${token.symbol} Price Chart`}
          src={`https://www.gmgn.cc/kline/solana/${tokenAddress}`}
          style={styles.chartFrame}
        />
      </div>

      <div style={styles.buttonContainer}>
        <button 
          style={{...styles.tradeBtn, ...styles.buyBtn}}
          onClick={() => openTradeModal('buy')}
        >
          Buy
        </button>
        <button 
          style={{...styles.tradeBtn, ...styles.sellBtn, ...(!userPosition ? styles.disabledBtn : {})}}
          onClick={() => openTradeModal('sell')}
          disabled={!userPosition}
        >
          Sell
        </button>
      </div>

      {showModal && (
        <TradeModal
          token={token}
          position={tradeMode === 'sell' ? userPosition : null}
          currentPrice={token.price} // Pass current price
          onClose={() => setShowModal(false)}
        />
      )}
    </div>
  );
};

const styles = {

  header: { marginBottom: '30px', textAlign: 'center' },
  title: { fontSize: '36px', fontWeight: 'bold', color: '#14f195', marginBottom: '10px' },
  subtitle: { fontSize: '14px', color: '#888', fontFamily: 'monospace' },
  link: { color: '#14f195', textDecoration: 'none' },

Step 4: Modify Search Result Click Behavior in the Navbar

Update frontend/src/components/Navbar.js so that clicking a search result also navigates to the new token page.

Replace the entire contents of frontend/src/components/Navbar.js with this code:

JavaScript

// FILE: frontend/src/components/Navbar.js

import React, { useState, useEffect, useRef } from 'react';
import { Link, useNavigate } from 'react-router-dom';
import { tokens } from '../api';
import TradeModal from './TradeModal';

const Navbar = ({ user, onLogout }) => {
  const [searchQuery, setSearchQuery] = useState('');
  const [searchResults, setSearchResults] = useState([]);
  const [showResults, setShowResults] = useState(false);
  const [searching, setSearching] = useState(false);
  const navigate = useNavigate(); // Use navigate hook
  const searchTimeoutRef = useRef(null);

  useEffect(() => {
    return () => { if (searchTimeoutRef.current) clearTimeout(searchTimeoutRef.current); };
  }, []);

  const handleSearch = (e) => {
    const query = e.target.value;
    setSearchQuery(query);
    if (searchTimeoutRef.current) clearTimeout(searchTimeoutRef.current);
    if (query.length < 2) { setSearchResults([]); setShowResults(false); setSearching(false); return; }
    setSearching(true);
    searchTimeoutRef.current = setTimeout(async () => {
      try { const res = await tokens.search(query); setSearchResults(res.data.pairs || []); setShowResults(true); } 
      catch (error) { console.error('Search error:', error); } 
      finally { setSearching(false); }
    }, 500);
  };
  
  const handleTokenClick = (token) => {
    setShowResults(false);
    setSearchQuery('');
    navigate(`/token/${token.tokenAddress}`, { state: { token } });
  };

  const handleSearchBlur = (e) => {
    if (!e.currentTarget.contains(e.relatedTarget)) {
      setShowResults(false);
    }
  };

  return (
    <nav style={styles.navbar}>
      <div style={styles.navLeft}><Link to="/" style={styles.logo}>🚀 PumpTrade</Link></div>
      {user && (
        <div style={styles.searchContainer} onBlur={handleSearchBlur}>
          <input type="text" placeholder="Search by address or symbol..." value={searchQuery} onChange={handleSearch} onFocus={() => searchQuery.length >= 2 && setShowResults(true)} style={styles.searchInput} />
          {searching && <div style={styles.searchLoading}>Searching...</div>}
          {showResults && (
            <div style={styles.searchResults}>
              {searchResults.length > 0 ? searchResults.slice(0, 10).map((token) => (
                <div key={token.tokenAddress} style={styles.searchResult} onMouseDown={() => handleTokenClick(token)}>
                  <div><div style={styles.tokenName}>{token.name} ({token.symbol})</div><div style={styles.tokenAddress}>{token.tokenAddress.slice(0, 8)}...</div></div>
                  <div style={styles.tokenPrice}>${token.price?.toFixed(8)}</div>
                </div>
              )) : <div style={styles.noResults}>No tokens found</div>}
            </div>
          )}
        </div>
      )}
      <div style={styles.navRight}>
        {user ? (<><div style={styles.balance}>Balance: {user.balance?.toFixed(2)} SOL</div><button onClick={onLogout} style={styles.logoutBtn}>Logout</button></>) : (<Link to="/login" style={styles.loginBtn}>Login</Link>)}
      </div>
    </nav>
  );
};

const styles = {
  navbar: { display: 'flex', justifyContent: 'space-between', alignItems: 'center', padding: '15px 30px', background: 'linear-gradient(135deg, #1a1a2e 0%, #16213e 100%)', borderBottom: '1px solid #14f195' },
  navLeft: { flex: 0.2 },
  logo: { fontSize: '28px', fontWeight: 'bold', color: '#14f195', textDecoration: 'none', textShadow: '0 0 10px rgba(20, 241, 149, 0.5)' },
  searchContainer: { flex: 0.6, position: 'relative' },
  searchInput: { width: '100%', padding: '12px 20px', background: 'rgba(255, 255, 255, 0.05)', border: '1px solid rgba(20, 241, 149, 0.3)', borderRadius: '25px', color: '#fff', fontSize: '14px', outline: 'none' },
  searchLoading: { position: 'absolute', top: '100%', left: 0, right: 0, marginTop: '10px', background: '#1a1a2e', border: '1px solid #14f195', borderRadius: '10px', padding: '15px', color: '#14f195', textAlign: 'center', zIndex: 1000 },
  searchResults: { position: 'absolute', top: '100%', left: 0, right: 0, marginTop: '10px', background: '#1a1a2e', border: '1px solid #14f195', borderRadius: '10px', maxHeight: '400px', overflowY: 'auto', zIndex: 1000 },
  searchResult: { display: 'flex', justifyContent: 'space-between', padding: '15px 20px', borderBottom: '1px solid rgba(20, 241, 149, 0.1)', cursor: 'pointer' },
  tokenName: { color: '#fff', fontWeight: 'bold', marginBottom: '5px' },
  tokenAddress: { color: '#888', fontSize: '12px' },
  tokenPrice: { color: '#14f195', fontWeight: 'bold' },
  noResults: { padding: '20px', textAlign: 'center', color: '#666' },
  navRight: { display: 'flex', alignItems: 'center', gap: '20px', flex: 0.2, justifyContent: 'flex-end' },
  balance: { color: '#14f195', fontWeight: 'bold', fontSize: '16px' },
  logoutBtn: { padding: '10px 25px', background: 'transparent', border: '2px solid #ff4757', borderRadius: '25px', color: '#ff4757', cursor: 'pointer', fontWeight: 'bold' },
