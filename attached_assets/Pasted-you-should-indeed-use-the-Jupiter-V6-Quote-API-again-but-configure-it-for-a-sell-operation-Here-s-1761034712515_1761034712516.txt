// --- NEW ROUTE for getting a SELL quote ---
router.post('/quote/sell', async (req, res) => {
  const { inputTokenAddress, tokenAmountLamports } = req.body; // Amount of the token TO SELL (in smallest unit)

  if (!inputTokenAddress || !tokenAmountLamports || tokenAmountLamports <= 0) {
    return res.status(400).json({ error: 'Missing or invalid parameters' });
  }

  const outputMint = 'So11111111111111111111111111111111111111112'; // SOL Address

  try {
    // --- JUPITER API CALL ---
    const quoteResponse = await axios.get('https://quote-api.jup.ag/v6/quote', {
      params: {
        inputMint: inputTokenAddress, // Token we are selling
        outputMint: outputMint,       // SOL we want to receive
        amount: tokenAmountLamports,  // Amount of the token to sell (smallest unit)
        slippageBps: 1000,            // Allow 10% slippage (adjust as needed)
        // onlyDirectRoutes: true,
        // platformFeeBps: 0
      }
    });

    const quoteData = quoteResponse.data;

     if (!quoteData || !quoteData.outAmount) {
        throw new Error('No quote available from Jupiter API for selling.');
     }

    // outAmount is SOL in lamports
    const solAmountOut = parseInt(quoteData.outAmount) / 1e9; // Convert lamports to SOL
    const effectivePriceSolPerToken = solAmountOut / (parseInt(tokenAmountLamports) / (10**(quoteData.inputMintDecimals || 9))); // Assume 9 decimals if unknown for input token

    res.json({
      tokenAmountInLamports: tokenAmountLamports,
      tokenAddressIn: inputTokenAddress,
      solAmountOut: solAmountOut, // SOL amount received
      solLamportsOut: quoteData.outAmount, // SOL amount in lamports
      effectivePriceSolPerToken: effectivePriceSolPerToken, // Average price received per token sold
      // Include other quote data if needed
    });

  } catch (error) {
    console.error("Jupiter Sell Quote API error:", error.response?.data || error.message);
    res.status(500).json({ error: 'Failed to get sell quote from Jupiter API' });
  }
});

import React, { useState, useEffect } from 'react';
import { trades, auth, tokens as tokenApi } from '../api'; // Import tokenApi

const TradeModal = ({ token, position, onClose }) => {
  // Assume position.amount is the number of tokens (e.g., 1,000,000)
  // Assume token decimals are needed - fetch or pass them in. Defaulting to 9 for now.
  const TOKEN_DECIMALS = token?.decimals || position?.token_decimals || 9;

  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  const [userBalance, setUserBalance] = useState(0);
  const [sellQuote, setSellQuote] = useState(null); // State to store the sell quote

  const isBuying = !position;
  const [buySolInput, setBuySolInput] = useState('1'); // SOL amount for buying
  const [buyQuote, setBuyQuote] = useState(null);     // State for buy quote

  // Fetch initial balance
  useEffect(() => {
    const loadUserBalance = async () => { /* ... as before ... */ };
    loadUserBalance();
  }, []);

  // --- Fetch Buy Quote ---
  useEffect(() => {
    if (isBuying && token && parseFloat(buySolInput) > 0) {
      const fetchQuote = async () => {
        try {
          setError(''); // Clear previous errors
          const res = await tokenApi.getBuyQuote(token.tokenAddress, parseFloat(buySolInput));
          setBuyQuote(res.data);
        } catch (err) {
          setError(err.response?.data?.error || 'Failed to get buy quote');
          setBuyQuote(null);
        }
      };
      // Debounce or add delay if needed
      const timer = setTimeout(fetchQuote, 300); // Add a small delay
      return () => clearTimeout(timer);
    } else {
      setBuyQuote(null);
    }
  }, [isBuying, token, buySolInput]);

  // --- Fetch Sell Quote ---
  useEffect(() => {
    if (!isBuying && position) {
      const fetchSellQuote = async () => {
        setLoading(true); // Indicate loading quote
        setError('');
        try {
          // Convert position amount to lamports based on decimals
          const amountLamports = Math.floor(position.amount * (10 ** TOKEN_DECIMALS));
          if (amountLamports <= 0) {
            throw new Error("Invalid position amount for quote.");
          }
          const res = await tokenApi.getSellQuote(position.token_address, amountLamports.toString());
          setSellQuote(res.data);
        } catch (err) {
          setError(err.response?.data?.error || 'Failed to get sell quote');
          setSellQuote(null);
        } finally {
          setLoading(false);
        }
      };
      fetchSellQuote();
    }
  }, [isBuying, position, TOKEN_DECIMALS]); // Depend on position and decimals

  // --- Handle Trade Execution ---
  const handleTrade = async () => {
    setLoading(true);
    setError('');
    try {
      if (isBuying) {
        // Buy Logic - Use buyQuote data
        if (!buyQuote || !buyQuote.tokenAmountOutLamports || !buyQuote.effectivePriceSol) {
          throw new Error("Buy quote is not available or invalid.");
        }
        if (parseFloat(buySolInput) > userBalance) {
           throw new Error("Insufficient balance.");
        }
         // Convert lamports back to token amount using decimals - IMPORTANT
        const tokenAmountFloat = parseFloat(buyQuote.tokenAmountOutLamports) / (10 ** TOKEN_DECIMALS);

        await trades.buy({
          tokenAddress: token.tokenAddress,
          tokenName: token.name,
          tokenSymbol: token.symbol,
          amount: tokenAmountFloat, // Use the quoted token amount
          price: buyQuote.effectivePriceSol, // Use the quoted effective price
          // Ensure backend uses the input SOL amount for sol_spent validation
          // You might need to adjust the backend /buy endpoint slightly
        });
        alert('Paper trade executed (Buy)!');

      } else {
        // Sell Logic - Use sellQuote data
        if (!sellQuote || !sellQuote.solLamportsOut) {
          throw new Error("Sell quote is not available or invalid.");
        }
        // Send positionId and the *quoted* SOL amount received to the backend
        await trades.sell({
          positionId: position.id,
          solReceived: sellQuote.solAmountOut, // Pass the SOL received from the quote
          // The backend /sell endpoint needs to be modified to accept solReceived
          // instead of just exitPrice.
        });
        alert('Paper trade executed (Sell)!');
      }
      onClose(); // Close modal on success
    } catch (error) {
      setError(error.message || error.response?.data?.error || 'Transaction failed');
    } finally {
      setLoading(false);
    }
  };

  const tokenData = token || position; // Use token for buy, position for sell info initially

  // --- Render Logic ---
  return (
    <div style={styles.overlay} onClick={onClose}>
      <div style={styles.modal} onClick={(e) => e.stopPropagation()}>
        <div style={styles.modalHeader}>
          <h2 style={styles.modalTitle}>{isBuying ? '🚀 Buy' : '💰 Sell'} {tokenData?.symbol || tokenData?.token_symbol}</h2>
          <button style={styles.closeBtn} onClick={onClose}>×</button>
        </div>

        <div style={styles.modalBody}>
          {isBuying && (
            <>
              <div style={styles.inputGroup}>
                <label style={styles.label}>Amount (SOL to Spend)</label>
                <input type="number" value={buySolInput} onChange={(e) => setBuySolInput(e.target.value)}
                       placeholder="1.0" style={styles.input} min="0" step="any" />
              </div>
              <div style={styles.summary}>
                <div style={styles.summaryRow}>
                  <span>Est. Tokens Received:</span>
                  <span style={styles.summaryValue}>
                    {buyQuote ? (parseFloat(buyQuote.tokenAmountOutLamports) / (10 ** TOKEN_DECIMALS)).toFixed(4) : 'Fetching...'} {tokenData?.symbol}
                  </span>
                </div>
                 <div style={styles.summaryRow}>
                  <span>Effective Price:</span>
                   <span style={styles.summaryValue}>
                    {buyQuote ? buyQuote.effectivePriceSol.toFixed(8) : 'Fetching...'} SOL/{tokenData?.symbol}
                  </span>
                </div>
                <div style={styles.summaryRow}><span>Your Balance:</span><span>{userBalance.toFixed(4)} SOL</span></div>
              </div>
            </>
          )}

          {!isBuying && position && (
             <>
               <div style={styles.inputGroup}>
                 <label style={styles.label}>Amount ({position.token_symbol} to Sell)</label>
                 {/* Display position amount - usually sell the whole position */}
                 <input type="number" value={position.amount.toString()} style={styles.input} readOnly />
               </div>
               <div style={styles.summary}>
                 <div style={styles.summaryRow}>
                   <span>Est. SOL Received:</span>
                   <span style={styles.summaryValue}>
                     {sellQuote ? sellQuote.solAmountOut.toFixed(4) : (loading ? 'Fetching...' : 'Error')} SOL

                    <span>Est. Profit/Loss:</span>

                  }}>

  );

// FILE: backend/routes/trades.js

// ... imports and other routes ...

router.post('/sell', authenticateToken, (req, res) => {
  // --- MODIFICATION START ---
  // Expect solReceived from the frontend quote instead of exitPrice directly
  const { positionId, solReceived } = req.body;

  if (!positionId || solReceived === undefined || solReceived === null || solReceived < 0) {
      return res.status(400).json({ error: 'Missing fields or invalid solReceived amount' });
  }
  // --- MODIFICATION END ---


  db.get('SELECT * FROM positions WHERE id = ? AND user_id = ?', [positionId, req.user.id], (err, position) => {
    if (err) return res.status(500).json({ error: 'Database error' });
    if (!position) return res.status(404).json({ error: 'Position not found' });

    // --- MODIFICATION START ---
    // Calculate effective exit price based on quoted solReceived
    const profitLoss = solReceived - position.sol_spent;
    const effectiveExitPrice = position.amount > 0 ? (solReceived / position.amount) : 0; // Avoid division by zero
    // --- MODIFICATION END ---

    db.serialize(() => {
      db.run('BEGIN TRANSACTION');
      // Update balance using the accurately quoted solReceived
      db.run('UPDATE users SET balance = balance + ?, total_profit = total_profit + ? WHERE id = ?',
        [solReceived, profitLoss, req.user.id], (err) => {
          if (err) { db.run('ROLLBACK'); return res.status(500).json({ error: 'Failed to update balance/profit' }); }

          // Record history using the calculated effective exit price
          db.run(`INSERT INTO trade_history (user_id, token_address, token_name, token_symbol, entry_price,
                  exit_price, amount, sol_spent, sol_received, profit_loss, opened_at)
                  VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
            [ req.user.id, position.token_address, position.token_name, position.token_symbol, position.entry_price,
              effectiveExitPrice, // Use calculated effective price for history
              position.amount, position.sol_spent,
              solReceived, // Use quoted SOL received for history
              profitLoss, position.opened_at ],
            (err) => {
              if (err) { db.run('ROLLBACK'); return res.status(500).json({ error: 'Failed to record trade history' }); }

              db.run('DELETE FROM positions WHERE id = ?', [positionId], (err) => {
                if (err) { db.run('ROLLBACK'); return res.status(500).json({ error: 'Failed to delete position' }); }

                db.run('COMMIT', (err) => {
                  if (err) { db.run('ROLLBACK'); return res.status(500).json({ error: 'Transaction commit failed' }); }

                  // Return updated user balance info
                   db.get('SELECT balance, total_profit FROM users WHERE id = ?', [req.user.id], (err, updatedUser) => {
                       if(err) {
                           // Log error but proceed with success response for the trade itself
                            console.error("Error fetching updated balance after sell:", err);
                             res.json({ message: 'Position closed successfully (balance fetch failed)', profitLoss, solReceived });
                       } else {
                            res.json({ message: 'Position closed', profitLoss, solReceived, newBalance: updatedUser.balance, newTotalProfit: updatedUser.total_profit });
                       }
                   });
                });
              });
            }
          );
        }
      );
    });
  });
});
