class DexDetector {
  /**
   * Identify DEX from instructions
   */
  identifyDex(instructions) {
    if (!instructions || !Array.isArray(instructions) || instructions.length === 0) {
      return 'Unknown';
    }

    // Check each instruction for known DEX programs
    for (const instruction of instructions) {
      if (!instruction || !instruction.programId) continue;
      
      const programId = instruction.programId;
      
      if (KNOWN_DEX_PROGRAMS[programId]) {
        return KNOWN_DEX_PROGRAMS[programId];
      }
    }

    return 'Unknown DEX';
  }

  /**
   * Check if any instruction uses a DEX program
   */
  hasDexProgram(instructions) {
    if (!instructions || !Array.isArray(instructions)) {
      return false;
    }

    return instructions.some(instruction => {
      if (!instruction || !instruction.programId) return false;
      return !!KNOWN_DEX_PROGRAMS[instruction.programId];
    });
  }

  /**
   * Check if specific program is a known DEX
   */
  isDexProgram(programId) {
    return !!KNOWN_DEX_PROGRAMS[programId];
  }

  /**
   * Get DEX name by program ID
   */
  getDexName(programId) {
    return KNOWN_DEX_PROGRAMS[programId] || 'Unknown';
  }

  /**
   * Get all known DEX programs
   */
  getAllKnownDexes() {
    return { ...KNOWN_DEX_PROGRAMS };
  }

  /**
   * Get DEX count in transaction
   */
  countDexInteractions(instructions) {
    if (!instructions || !Array.isArray(instructions)) {
      return 0;
    }

    return instructions.filter(i => 
      i && i.programId && this.isDexProgram(i.programId)
    ).length;
  }
}

module.exports = new DexDetector();
