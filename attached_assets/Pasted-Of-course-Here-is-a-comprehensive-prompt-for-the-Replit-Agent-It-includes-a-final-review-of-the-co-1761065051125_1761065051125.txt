router.get('/search', async (req, res) => {
    const { query } = req.query;
    if (!query || query.length < 2) return res.status(400).json({ error: 'Query min 2 chars' });
    try {
        const response = await fetchWithRetry(`https://api.dexscreener.com/latest/dex/search?q=${encodeURIComponent(query)}`);
        const pumpPairs = (response.data.pairs || []).filter(p => p.dexId === 'pump').map(p => ({
            tokenAddress: p.baseToken.address, name: p.baseToken.name || 'Unknown', symbol: p.baseToken.symbol || '???',
            price: parseFloat(p.priceUsd || 0), marketCap: p.fdv || 0, creator: 'N/A'
        }));
        res.json({ pairs: pumpPairs });
    } catch (error) {
        console.error('Search error:', error.message);
        res.status(500).json({ error: 'Search failed', pairs: [] });
    }
});

router.post('/quote/buy', async (req, res) => {
    const { outputTokenAddress, solAmount } = req.body;
    if (!outputTokenAddress || !solAmount || solAmount <= 0) return res.status(400).json({ error: 'Missing params' });
    const lamportsAmount = Math.floor(solAmount * 1e9);
    try {
        const { data } = await axios.get('https://quote-api.jup.ag/v6/quote', {
            params: { inputMint: 'So11111111111111111111111111111111111111112', outputMint: outputTokenAddress, amount: lamportsAmount, slippageBps: 1000 }
        });
        if (!data || !data.outAmount) throw new Error('No Jupiter quote available.');
        const tokenDecimals = data.outputMintInfo?.decimals || 9;
        const effectivePriceSol = solAmount / (parseInt(data.outAmount) / (10 ** tokenDecimals));
        res.json({ solAmountIn: solAmount, tokenAmountOutLamports: data.outAmount, tokenAddressOut: outputTokenAddress, effectivePriceSol: effectivePriceSol, decimals: tokenDecimals });

// FILE: frontend/src/components/PositionsBar.js

import React, { useState, useEffect } from 'react';
import { trades } from '../api';
import TradeModal from './TradeModal';
import { useWebSocket } from '../App'; // Import the context hook

const PositionsBar = ({ refreshUser }) => {
  const [positions, setPositions] = useState([]);
  const [selectedPosition, setSelectedPosition] = useState(null);
  const [showModal, setShowModal] = useState(false);
  const wsTokens = useWebSocket(); // Get live token data from WebSocket context

  useEffect(() => {
    const loadPositions = async () => {
      try {
        const res = await trades.getPositions();
        setPositions(res.data.positions || []);
      } catch (error) {
        console.error('Error loading positions:', error);
      }
    };
    loadPositions();
    const interval = setInterval(loadPositions, 60000); // Refresh positions list every minute
    return () => clearInterval(interval);
  }, []);

  const calculatePL = (position) => {
    const allLiveTokens = [...wsTokens.new, ...wsTokens.graduating, ...wsTokens.graduated];
    const liveTokenData = allLiveTokens.find(t => t.tokenAddress === position.token_address);

    const currentPrice = liveTokenData ? liveTokenData.price : position.entry_price;
    const priceAvailable = !!liveTokenData;

    const currentValue = position.amount * currentPrice;
    const pl = currentValue - position.sol_spent;
    const plPercent = position.sol_spent > 0 ? ((currentValue / position.sol_spent - 1) * 100) : 0;
    
    return { pl, plPercent: plPercent.toFixed(2), currentPrice, priceAvailable };
  };

  const handleSell = (position) => {
    setSelectedPosition(position);
    setShowModal(true);
  };

  const handleModalClose = () => {
    setShowModal(false);
    setSelectedPosition(null);
    if (refreshUser) refreshUser();
  };

  return (
    <div style={styles.positionsBar}>
      <div style={styles.header}>
        <h3 style={styles.title}>📊 My Positions</h3>
        <div style={styles.count}>{positions.length}</div>
      </div>
      <div style={styles.positionsList}>
        {positions.length === 0 ? (
          <div style={styles.empty}>No open positions</div>
        ) : (
          positions.map((pos) => {
            const { pl, plPercent, currentPrice, priceAvailable } = calculatePL(pos);
            const isProfit = pl >= 0;
            return (
              <div key={pos.id} style={styles.position}>
                <div style={styles.positionHeader}>
                  <div style={styles.tokenName}>{pos.token_symbol}</div>
                  <div style={{ ...styles.pl, color: priceAvailable ? (isProfit ? '#14f195' : '#ff4757') : '#888' }}>
                    {priceAvailable ? `${isProfit ? '+' : ''}${plPercent}%` : 'N/A'}
                  </div>
                </div>
                <div style={styles.positionDetails}>
                  <div style={styles.detailRow}><span>Entry Price:</span><span>{pos.entry_price.toFixed(8)} SOL</span></div>
                  <div style={styles.detailRow}><span>Current Price:</span><span>{priceAvailable ? currentPrice.toFixed(8) : 'N/A'} SOL</span></div>
                  <div style={styles.detailRow}><span>P/L:</span><span style={{color: priceAvailable ? (isProfit ? '#14f195' : '#ff4757') : '#888'}}>{priceAvailable ? `${isProfit ? '+' : ''}${pl.toFixed(4)} SOL` : 'N/A'}</span></div>
                </div>
                <button onClick={() => handleSell(pos)} style={styles.sellBtn}>Sell</button>
              </div>
            );
          })
        )}
      </div>
      {showModal && <TradeModal position={selectedPosition} currentPrice={calculatePL(selectedPosition).currentPrice} onClose={handleModalClose} />}
    </div>
  );
};

Step 3: Important Architectural Notes (Final Review)

Before creating the website, please review the following architectural points about the code. While the tool is functional for its educational purpose, these are critical considerations for any real-world application.

    Critical Issue: Floating-Point Math for Currency

        Problem: The backend database still uses the REAL (floating-point) data type for all financial values. This will lead to small precision errors that accumulate over time, corrupting user balances and P/L calculations.

        Proper Solution (for future reference): All currency values should be stored and calculated as integers (e.g., using Solana's smallest unit, Lamports, where 1 SOL = 1,000,000,000 Lamports).

    Critical Security Flaw: JWT Storage in localStorage

        Problem: The frontend stores the user's authentication token in the browser's localStorage. This is vulnerable to Cross-Site Scripting (XSS) attacks, which could allow an attacker to steal the token and take over a user's account.

        Proper Solution (for future reference): The authentication system should be refactored to store the JWT in a secure, HttpOnly cookie, which cannot be accessed by client-side JavaScript.
