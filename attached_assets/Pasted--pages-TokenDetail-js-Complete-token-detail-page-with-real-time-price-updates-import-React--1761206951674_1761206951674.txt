// pages/TokenDetail.js
// Complete token detail page with real-time price updates

import React, { useState, useEffect } from 'react';
import { useParams } from 'react-router-dom';
import useRealtimePrice from '../hooks/useRealtimePrice';
import TokenChartWithVolume from '../components/TokenChartWithVolume';
import axios from 'axios';

const TokenDetail = () => {
  const { address } = useParams(); // Get token address from URL
  const [tokenInfo, setTokenInfo] = useState(null);
  const [loading, setLoading] = useState(true);
  
  // Real-time price hook (updates every 5 seconds automatically)
  const { 
    price, 
    priceChange, 
    loading: priceLoading, 
    lastUpdate 
  } = useRealtimePrice(address, 5000); // 5 second updates

  // Fetch token metadata
  useEffect(() => {
    const fetchTokenInfo = async () => {
      try {
        const response = await axios.get(
          `https://api.dexscreener.com/latest/dex/tokens/${address}`
        );

        if (response.data?.pairs?.[0]) {
          const pair = response.data.pairs[0];
          setTokenInfo({
            name: pair.baseToken.name,
            symbol: pair.baseToken.symbol,
            address: address,
            volume24h: parseFloat(pair.volume?.h24 || 0),
            liquidity: parseFloat(pair.liquidity?.usd || 0),
            priceChange24h: parseFloat(pair.priceChange?.h24 || 0),
            marketCap: parseFloat(pair.fdv || 0),
            pairAddress: pair.pairAddress,
            dexId: pair.dexId,
            url: pair.url
          });
        }
        setLoading(false);
      } catch (err) {
        console.error('Error fetching token info:', err);
        setLoading(false);
      }
    };

    if (address) {
      fetchTokenInfo();
    }
  }, [address]);

  if (loading) {
    return (
      <div className="token-detail-loading">
        <div className="spinner"></div>
        <p>Loading token details...</p>
      </div>
    );
  }

  if (!tokenInfo) {
    return (
      <div className="token-detail-error">
        <h2>Token not found</h2>
        <p>The token address may be invalid or not yet indexed.</p>
      </div>
    );
  }

  return (
    <div className="token-detail-page">
      {/* Header with Real-Time Price */}
      <div className="token-header">
        <div className="token-info">
          <h1>{tokenInfo.name}</h1>
          <span className="token-symbol">{tokenInfo.symbol}</span>
        </div>

        <div className="token-price-section">
          <div className="current-price">
            {priceLoading ? (
              <span className="loading-price">Loading...</span>
            ) : (
              <>
                <span className="price-value">
                  ${price?.toFixed(8) || '0.00000000'}
                </span>
                {priceChange !== 0 && (
                  <span className={`price-change ${priceChange >= 0 ? 'positive' : 'negative'}`}>
                    {priceChange >= 0 ? '▲' : '▼'} {Math.abs(priceChange).toFixed(2)}%
                  </span>
                )}
              </>
            )}
          </div>
          {lastUpdate && (
            <div className="last-update">
              Last updated: {lastUpdate.toLocaleTimeString()}
              <span className="live-indicator">🔴 LIVE</span>
            </div>
          )}
        </div>
      </div>

      {/* Token Stats Grid */}
      <div className="token-stats-grid">
        <div className="stat-card">
          <div className="stat-label">24h Change</div>
          <div className={`stat-value ${tokenInfo.priceChange24h >= 0 ? 'positive' : 'negative'}`}>
            {tokenInfo.priceChange24h >= 0 ? '+' : ''}{tokenInfo.priceChange24h.toFixed(2)}%
          </div>
        </div>

        <div className="stat-card">
          <div className="stat-label">24h Volume</div>
          <div className="stat-value">
            ${tokenInfo.volume24h.toLocaleString(undefined, { maximumFractionDigits: 0 })}
          </div>
        </div>

        <div className="stat-card">
          <div className="stat-label">Liquidity</div>
          <div className="stat-value">
            ${tokenInfo.liquidity.toLocaleString(undefined, { maximumFractionDigits: 0 })}
          </div>
        </div>

        <div className="stat-card">
          <div className="stat-label">Market Cap</div>
          <div className="stat-value">
            ${tokenInfo.marketCap.toLocaleString(undefined, { maximumFractionDigits: 0 })}
          </div>
        </div>
      </div>

      {/* Price Chart with Volume */}
      <TokenChartWithVolume tokenAddress={address} />

      {/* Trading Actions */}
      <div className="trading-actions">
        <button className="btn-buy" onClick={() => handleBuy()}>
          Buy {tokenInfo.symbol}
        </button>
        <button className="btn-sell" onClick={() => handleSell()}>
          Sell {tokenInfo.symbol}
        </button>
      </div>

      {/* Token Details */}
      <div className="token-details-section">
        <h3>Token Information</h3>
        <div className="detail-row">
          <span className="detail-label">Contract Address:</span>
          <span className="detail-value">
            {address}
            <button 
              className="copy-btn" 
              onClick={() => navigator.clipboard.writeText(address)}
              title="Copy address"
            >
              📋
            </button>
          </span>
        </div>
        <div className="detail-row">
          <span className="detail-label">Pair Address:</span>
          <span className="detail-value">{tokenInfo.pairAddress}</span>
        </div>
        <div className="detail-row">
          <span className="detail-label">DEX:</span>
          <span className="detail-value">{tokenInfo.dexId}</span>
        </div>
        <div className="detail-row">
          <span className="detail-label">View on DexScreener:</span>
          <a 
            href={tokenInfo.url} 
            target="_blank" 
            rel="noopener noreferrer"
            className="external-link"
          >
            Open →
          </a>
        </div>
      </div>
    </div>
  );

  function handleBuy() {
    // Your existing buy logic
    console.log('Buy clicked');
  }

  function handleSell() {
    // Your existing sell logic
    console.log('Sell clicked');
  }
};

export default TokenDetail;
