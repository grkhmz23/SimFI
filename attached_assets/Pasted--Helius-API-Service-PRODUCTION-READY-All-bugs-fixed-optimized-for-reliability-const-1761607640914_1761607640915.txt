/**
 * Helius API Service - PRODUCTION READY
 * All bugs fixed, optimized for reliability
 */

const fetch = require('node-fetch');
const { HELIUS_API_KEY, HELIUS_BASE_URL } = require('../config/helius.config');

class HeliusService {
  constructor() {
    this.apiKey = HELIUS_API_KEY;
    this.baseUrl = HELIUS_BASE_URL;
    this.requestCount = 0;
    this.lastResetTime = Date.now();
  }

  /**
   * Generic request handler with comprehensive error handling
   */
  async makeRequest(url, options = {}, retries = 3) {
    for (let attempt = 1; attempt <= retries; attempt++) {
      try {
        // Validate API key
        if (!this.apiKey || this.apiKey === '') {
          throw new Error('HELIUS_API_KEY not configured. Add it to environment variables.');
        }

        const controller = new AbortController();
        const timeout = setTimeout(() => controller.abort(), 30000); // 30s timeout

        const response = await fetch(url, {
          ...options,
          signal: controller.signal,
          headers: {
            'Content-Type': 'application/json',
            ...options.headers
          }
        });

        clearTimeout(timeout);

        // Handle rate limiting
        if (response.status === 429) {
          const retryAfter = response.headers.get('Retry-After') || 5;
          console.warn(`Rate limited. Retrying after ${retryAfter}s...`);
          await this.sleep(retryAfter * 1000);
          continue;
        }

        // Handle errors
        if (!response.ok) {
          const errorText = await response.text();
          throw new Error(`Helius API error ${response.status}: ${errorText.slice(0, 200)}`);
        }

        this.requestCount++;
        
        // Parse JSON safely
        const text = await response.text();
        if (!text) {
          return null;
        }

        try {
          return JSON.parse(text);
        } catch (e) {
          console.error('JSON parse error:', text.slice(0, 100));
          throw new Error('Invalid JSON response from Helius');
        }

      } catch (error) {
        console.error(`Request attempt ${attempt}/${retries} failed:`, error.message);
        
        // Don't retry on certain errors
        if (error.message.includes('not configured') || 
            error.message.includes('401') ||
            error.message.includes('403')) {
          throw error;
        }

        if (attempt === retries) {
          throw error;
        }

        // Exponential backoff
        await this.sleep(Math.min(Math.pow(2, attempt) * 1000, 10000));
      }
    }
  }

  /**
   * Fetch token metadata - FIXED
   */
  async fetchTokenMetadata(mintAddress) {
    // Validate input
    if (!mintAddress || typeof mintAddress !== 'string') {
      throw new Error('Invalid mint address');
    }

    const url = `${this.baseUrl}/v0/token-metadata?api-key=${this.apiKey}`;
    
    try {
      const response = await this.makeRequest(url, {
        method: 'POST',
        body: JSON.stringify({
          mintAccounts: [mintAddress],
          includeOffChain: true,
          disableCache: false
        })
      });

      if (!response || response.length === 0) {
        throw new Error('Token not found or invalid mint address');
      }

      const tokenData = response[0];
      
      // ✅ FIX: Safe property access with fallbacks
      const onChainData = tokenData.onChainAccountInfo?.data?.parsed?.info || {};
      const metadata = tokenData.onChainMetadata?.metadata?.data || {};
      const offChain = tokenData.offChainMetadata?.metadata || {};
      
      return {
        name: metadata.name?.trim() || offChain.name || 'Unknown Token',
        symbol: metadata.symbol?.trim() || offChain.symbol || 'N/A',
        supply: onChainData.supply || '0',
        decimals: onChainData.decimals ?? 9, // ✅ FIX: Use nullish coalescing
        mintAuthority: onChainData.mintAuthority || null,
        freezeAuthority: onChainData.freezeAuthority || null,
        logoUri: offChain.image || null,
        isInitialized: onChainData.isInitialized || false
      };

    } catch (error) {
      console.error('Error fetching token metadata:', error);
      throw new Error(`Failed to fetch token metadata: ${error.message}`);
    }
  }

  /**
   * Fetch ALL token transactions - FIXED PAGINATION
   */
  async fetchAllTokenTransactions(mintAddress, maxTransactions = 1000) {
    const allTransactions = [];
    let beforeSignature = null;
    let loopDetection = new Set(); // ✅ FIX: Detect infinite loops
    let emptyResponses = 0;

    try {
      console.log(`📡 Fetching transactions for ${mintAddress}...`);

      while (allTransactions.length < maxTransactions) {
        // ✅ FIX: Correct Helius pagination
        let url = `${this.baseUrl}/v0/addresses/${mintAddress}/transactions?api-key=${this.apiKey}`;
        
        if (beforeSignature) {
          // ✅ FIX: Detect loop
          if (loopDetection.has(beforeSignature)) {
            console.warn('⚠️  Pagination loop detected, stopping');
            break;
          }
          loopDetection.add(beforeSignature);
          url += `&before=${beforeSignature}`;
        }

        const batch = await this.makeRequest(url);

        // ✅ FIX: Handle empty responses
        if (!batch || !Array.isArray(batch) || batch.length === 0) {
          emptyResponses++;
          if (emptyResponses >= 2) {
            console.log('No more transactions available');
            break;
          }
          continue;
        }

        emptyResponses = 0; // Reset counter
        allTransactions.push(...batch);
        
        // ✅ FIX: Get last signature for pagination
        const lastTx = batch[batch.length - 1];
        const newBeforeSignature = lastTx?.signature;
        
        // ✅ FIX: Check if we got a new signature
        if (!newBeforeSignature || newBeforeSignature === beforeSignature) {
          console.log('Reached end of transaction history');
          break;
        }
        
        beforeSignature = newBeforeSignature;

        console.log(`   Fetched ${allTransactions.length} transactions...`);

        // ✅ FIX: Rate limiting (100 req/min on free tier)
        await this.sleep(700); // ~85 req/min to be safe

        // ✅ FIX: Safety check for unusually long histories
        if (loopDetection.size > 50) {
          console.warn('⚠️  Too many pages, stopping for safety');
          break;
        }
      }

      console.log(`✅ Total transactions fetched: ${allTransactions.length}`);
      return allTransactions;

    } catch (error) {
      console.error('Error fetching transactions:', error);
      
      // ✅ FIX: Return partial results instead of failing completely
      if (allTransactions.length > 0) {
        console.log(`⚠️  Returning ${allTransactions.length} partial results`);
        return allTransactions;
      }
      
      throw error;
    }
  }

  /**
   * Health check
   */
  async healthCheck() {
    try {
      const url = `${this.baseUrl}/v0/token-metadata?api-key=${this.apiKey}`;
      await this.makeRequest(url, {
        method: 'POST',
        body: JSON.stringify({
          mintAccounts: ['So11111111111111111111111111111111111111112'] // Wrapped SOL
        })
      }, 1);
      return true;
    } catch (error) {
      console.error('Helius health check failed:', error.message);
      return false;
    }
  }

  /**
   * Get rate limit info
   */
  getRateLimitInfo() {
    const now = Date.now();
    const timeSinceReset = now - this.lastResetTime;
    
    // Reset counter every minute
    if (timeSinceReset > 60000) {
      this.requestCount = 0;
      this.lastResetTime = now;
    }

    return {
      requestsMade: this.requestCount,
      resetIn: Math.ceil((60000 - timeSinceReset) / 1000),
      maxPerMinute: 100,
      percentUsed: (this.requestCount / 100) * 100
    };
  }

  /**
   * Sleep utility
   */
  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

module.exports = new HeliusService();
