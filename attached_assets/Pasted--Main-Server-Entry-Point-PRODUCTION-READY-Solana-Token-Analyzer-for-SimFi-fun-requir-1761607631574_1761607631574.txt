/**
 * Main Server Entry Point - PRODUCTION READY
 * Solana Token Analyzer for SimFi.fun
 */

require('dotenv').config();
const express = require('express');
const cors = require('cors');
const path = require('path');

// Middleware
const errorHandler = require('./middleware/errorHandler');
const corsConfig = require('./middleware/cors');
const { analyzeTokenLimiter, statusCheckLimiter } = require('./middleware/rateLimiter');

// Routes
const analyzerRoutes = require('./routes/tokenAnalyzer');

// Initialize app
const app = express();
const PORT = process.env.PORT || 3000;

// Trust proxy (for Replit/Cloud deployments)
app.set('trust proxy', 1);

// Middleware setup (ORDER MATTERS!)
app.use(corsConfig); // CORS must be first
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true, limit: '10mb' }));

// Request logging (development only)
if (process.env.NODE_ENV !== 'production') {
  app.use((req, res, next) => {
    console.log(`${new Date().toISOString()} - ${req.method} ${req.path}`);
    next();
  });
}

// Health check endpoint
app.get('/health', (req, res) => {
  res.json({ 
    status: 'ok', 
    timestamp: new Date().toISOString(),
    uptime: process.uptime()
  });
});

// API Routes with rate limiting
app.use('/api/analyze-token', analyzeTokenLimiter);
app.use('/api', analyzerRoutes);

// Serve static files (frontend) - only in production
if (process.env.NODE_ENV === 'production') {
  app.use(express.static(path.join(__dirname, '../client/build')));
  
  // Catch-all route for SPA
  app.get('*', (req, res) => {
    res.sendFile(path.join(__dirname, '../client/build/index.html'));
  });
}

// Global error handler (MUST BE LAST)
app.use(errorHandler);

// Handle uncaught exceptions
process.on('uncaughtException', (error) => {
  console.error('Uncaught Exception:', error);
  process.exit(1);
});

// Handle unhandled promise rejections
process.on('unhandledRejection', (reason, promise) => {
  console.error('Unhandled Rejection at:', promise, 'reason:', reason);
  process.exit(1);
});

// Start server
const server = app.listen(PORT, () => {
  console.log(`🚀 Server running on port ${PORT}`);
  console.log(`📊 Token Analyzer API: http://localhost:${PORT}/api`);
  console.log(`🌍 Environment: ${process.env.NODE_ENV || 'development'}`);
  
  // Validate environment
  if (!process.env.HELIUS_API_KEY) {
    console.error('❌ CRITICAL: HELIUS_API_KEY not set!');
    console.error('   Add it to Replit Secrets or .env file');
  } else {
    console.log('✅ Helius API configured');
  }
});

// Graceful shutdown
const gracefulShutdown = () => {
  console.log('\n🛑 Shutting down gracefully...');
  
  server.close(() => {
    console.log('✅ Server closed');
    process.exit(0);
  });

  // Force close after 10 seconds
  setTimeout(() => {
    console.error('⚠️  Forced shutdown');
    process.exit(1);
  }, 10000);
};

process.on('SIGTERM', gracefulShutdown);
process.on('SIGINT', gracefulShutdown);

module.exports = app;
